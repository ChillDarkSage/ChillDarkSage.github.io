C#中的XML文件处理

Date: Oct 4, 2021  15:59:23 Category: 杂项

  初步学习Xamarin时, 遇到了需要存储应用数据的情况; 早听说过XML的大名, 于是决定使用XML文件进行相关操作.

  下面是一些读写操作的示例.

/// <summary>
/// 读入Xml数据
/// </summary>
public List<User> GetUsers()
{
    List<User> users = new List<User>();

    XmlDocument document = new XmlDocument();
    document.Load("Users.xml");

    XmlNodeList userNodeList = document.SelectNodes("User");
    if (userNodeList.Count == 0)
        return users;
    foreach (XmlElement e in userNodeList)
    {
        string name = e.GetAttributeNode("Name").InnerText;
        int id = int.Parse(e.GetAttribute("Id"));
        string sex = e.GetAttribute("Sex");

        users.Add(new User(name, id, sex));
    }

    return users;
} 
/// <summary>
/// 增加User结点
/// </summary>
public void AddUser(User user)
{
    XmlDocument document = new XmlDocument();
    document.Load("Users.xml");

    XmlNode root = document.SelectSingleNode("Users");

    XmlElement userToAdd = document.CreateElement("User");
    XmlElement nameNode = document.CreateElement("Name");
    nameNode.InnerText = user.Name;
    userToAdd.AppendChild(nameNode);
    userToAdd.SetAttribute("Id", user.Id.ToString());
    userToAdd.SetAttribute("Sex", user.Sex);

    root.AppendChild(userToAdd);
    document.Save("Users.xml");
}
/// <summary>
/// 删除User结点
/// </summary>
public void DelUser(User user)
{
    XmlDocument document = new XmlDocument();
    document.Load("Users.xml");

    XmlNodeList userNodeList = document.SelectNodes("User");

    foreach (XmlElement e in userNodeList)
    {
        if (e.GetAttributeNode("Name").InnerText == user.Name
            && int.Parse(e.GetAttribute("Id")) == user.Id
            && e.GetAttribute("Sex") == user.Sex)
        {
            e.RemoveAll();
            break;
        }
    }

    document.Save("Users.xml");
}
/// <summary>
/// 编辑User结点
/// </summary>
public void EditUser(User oldUserInfo, User newUserInfo)
{
    XmlDocument document = new XmlDocument();
    document.Load("Users.xml");

    XmlNodeList userNodeList = document.SelectSingleNode("Users").ChildNodes;
    foreach (XmlElement e in userNodeList)
    {
        if (e.GetAttributeNode("Name").InnerText == oldUserInfo.Name
            && int.Parse(e.GetAttribute("Id")) == oldUserInfo.Id
            && e.GetAttribute("Sex") == oldUserInfo.Sex)
        {
            e.ChildNodes[0].InnerText = newUserInfo.Name;
            e.SetAttribute("Id", newUserInfo.Id.ToString());
            e.SetAttribute("Sex", newUserInfo.Sex);
            break;
        }
    }

    document.Save(XMLPath);
}
/// <summary>
/// 通过Id获取User
/// </summary>
public User GetUser(int Id)
{
    XmlDocument document = new XmlDocument();
    document.Load("Users.xml");

    XmlNodeList userNodeList = document.SelectNodes("User");
    foreach (XmlElement e in userNodeList)
        if (int.Parse(e.GetAttribute("Id")) == Id)
            return new User(e.GetAttributeNode("Name").InnerText, int.Parse(e.GetAttribute("Id")), e.GetAttribute("Sex"));

    return null;
}

public class User
{
    public string Name { get; set; }

    public int Id { get; set; }

    public string Sex { get; set; }

    public User(string name, int id, string sex)
    {
        Name = name; 
        Id = id;
        Sex = sex;
    }
}
