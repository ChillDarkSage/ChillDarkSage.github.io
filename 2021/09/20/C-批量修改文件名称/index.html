C#批量修改文件名称

Date: Sep 20, 2021  16:34:24 Category: 小工具

  在B站上下载了一堆视频打算转成音频, 但是发现所有视频文件都包含“_高清 720P”这个后缀, 看起来非常难受. 手动改则会耗费非常多的时间, 所以写了个小工具自动改名, 原理不能再简单: 只是删除所有文件名中这个字符串而已, 没什么好说的.

  新建一个WPF程序(使用了FluentWPF的UI, 还不错).

<fw:AcrylicWindow x:Class="Rename.MainWindow"
        xmlns="http://schemas.microsoft.com/winfx/2006/xaml/presentation"
        xmlns:x="http://schemas.microsoft.com/winfx/2006/xaml"
        xmlns:d="http://schemas.microsoft.com/expression/blend/2008"
        xmlns:mc="http://schemas.openxmlformats.org/markup-compatibility/2006"
        xmlns:fw="clr-namespace:SourceChord.FluentWPF;assembly=FluentWPF"
        mc:Ignorable="d"
        Title="改名" Height="150" Width="300" ResizeMode="CanMinimize"  WindowStartupLocation="CenterScreen">
    <StackPanel >
        <TextBox x:Name="textBoxPath" IsReadOnly="True" Width="200" Height="30" FontSize="10" Foreground="WhiteSmoke" Style="{StaticResource TextBoxRevealStyle}" HorizontalAlignment="Center" Margin="5"/>
        <Button Content="浏览" Width="200" Height="30" FontSize="10" Foreground="WhiteSmoke" Style="{StaticResource ButtonAccentRevealStyle}" HorizontalAlignment="Center" Click="Select"/>
        <Button Content="工作" Width="200" Height="30" FontSize="10" Foreground="WhiteSmoke" Style="{StaticResource ButtonAccentRevealStyle}" HorizontalAlignment="Center" Margin="3" Click="Work" />
    </StackPanel>
</fw:AcrylicWindow>

  后台代码如下:

public partial class MainWindow
{
    public MainWindow()
    {
        InitializeComponent();
    }

    private void Select(object sender, RoutedEventArgs e)
    {
        var fbd = new System.Windows.Forms.FolderBrowserDialog();
        if (fbd.ShowDialog() == System.Windows.Forms.DialogResult.OK)
        {
            this.textBoxPath.Text = fbd.SelectedPath;
        }
    }

    private void Work(object sender, RoutedEventArgs e)
    {
        if (this.textBoxPath.Text != null)
        {
            var path = this.textBoxPath.Text;
            var files = Directory.GetFiles(path);
            if (files.Length != 0)
            {
                for (int i = 0; i < files.Length; i++)
                {
                    string name = files[i];
                    if (Regex.IsMatch(name, @"_高清 720P"))
                    {
                        var newName = Regex.Replace(name, @"_高清 720P", "");
                        File.Move(name, newName);
                    }
                }
                MessageBox.Show("Done.");
            }
        }
    }
}
